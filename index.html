import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; // Headers obligatorios para permitir peticiones desde GitHub Pages (CORS) const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, GET, OPTIONS", }; const BATCH_SIZE = 500; // tamaño de lote para no mandar arreglos gigantes en un solo upsert // Hace upsert en lotes contra una tabla, con onConflict explícito, y devuelve // { insertados, errores } en vez de tragarse los fallos. async function upsertEnLotes( supabase: any, tabla: string, registros: any[], onConflictCol: string ) { const errores: any[] = []; let insertados = 0; // Separar los registros que no tienen la columna de conflicto (no se pueden // deduplicar de forma confiable con onConflict, ya que NULL != NULL en Postgres) const validos = registros.filter((r) => r?.[onConflictCol] != null && r?.[onConflictCol] !== ""); const sinClave = registros.length - validos.length; for (let i = 0; i < validos.length; i += BATCH_SIZE) { const lote = validos.slice(i, i + BATCH_SIZE); const { error, count } = await supabase .from(tabla) .upsert(lote, { onConflict: onConflictCol, ignoreDuplicates: false, count: "exact" }); if (error) { errores.push({ tabla, lote_desde: i, mensaje: error.message, detalles: error.details ?? null }); } else { insertados += lote.length; } } return { tabla, insertados, sinClave, errores }; } // Hace fetch a una API externa y NUNCA se traga el error en silencio: // devuelve { data, diagnostico } donde diagnostico explica qué pasó si data quedó vacío. async function fetchApiConDiagnostico(nombre: string, url: string, opciones: RequestInit = {}) { try { const r = await fetch(url, opciones); const textoCrudo = await r.text(); if (!r.ok) { return { data: [], diagnostico: { fuente: nombre, ok: false, http_status: r.status, mensaje: `La API respondió HTTP ${r.status}`, cuerpo_respuesta: textoCrudo.slice(0, 500), // recortado para no inflar el log }, }; } let json: any; try { json = JSON.parse(textoCrudo); } catch (parseErr: any) { return { data: [], diagnostico: { fuente: nombre, ok: false, http_status: r.status, mensaje: `Respuesta no es JSON válido: ${parseErr.message}`, cuerpo_respuesta: textoCrudo.slice(0, 500), }, }; } return { data: json, diagnostico: { fuente: nombre, ok: true, http_status: r.status } }; } catch (err: any) { // Esto captura fallos de red reales: DNS, timeout, conexión rechazada, etc. return { data: [], diagnostico: { fuente: nombre, ok: false, http_status: null, mensaje: `Fallo de red/fetch: ${err.message}`, cuerpo_respuesta: null, }, }; } } Deno.serve(async (req) => { // 1. Responder inmediatamente a las peticiones Preflight de los navegadores if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } try { // 2. Cliente Supabase con Service Role o Credenciales de Entorno const supabaseUrl = Deno.env.get("SUPABASE_URL") ?? ""; const serviceKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? ""; const anonKey = Deno.env.get("SUPABASE_ANON_KEY") ?? ""; const supabaseKey = serviceKey || anonKey; const usandoAnonFallback = !serviceKey && !!anonKey; if (!supabaseUrl || !supabaseKey) { return new Response( JSON.stringify({ status: "error", message: "Faltan variables de entorno SUPABASE_URL y/o SUPABASE_SERVICE_ROLE_KEY.", }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } const supabase = createClient(supabaseUrl, supabaseKey); // Formatear fechas const hoy = new Date().toISOString().split("T")[0]; const hace7dias = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000).toISOString().split("T")[0]; // 3. Peticiones concurrentes en paralelo a las 3 APIs, con diagnóstico si algo falla const [leadsRes, historialesRes, llamadasRes] = await Promise.all([ fetchApiConDiagnostico( "leads", "https://prospektia.red.com.sv/api/external/leads-calificados", { method: "GET", headers: { "X-API-Key": "RedApi_2026_SuperSegura_9XK2" } } ), fetchApiConDiagnostico( "historiales", "https://kpi.red.com.sv/api/historial-llamadas-ejecutivos", { method: "GET", headers: { "X-API-KEY": "mso_papi_2026_7f2b9c4d8e6a1b3f5d7c9e0a2b4c6d8f" } } ), fetchApiConDiagnostico( "llamadas", `https://api.red.com.sv/idata/api/v1/desaIssabel/getResumenLlamada?fechaInicial=${hace7dias}&fechaFinal=${hoy}` ), ]); // Extraer arreglos (algunas APIs devuelven el arreglo directo, otras { data: [...] }) const dataLeads = Array.isArray(leadsRes.data) ? leadsRes.data : leadsRes.data?.data || []; const dataHistoriales = Array.isArray(historialesRes.data) ? historialesRes.data : historialesRes.data?.data || []; const dataLlamadas = Array.isArray(llamadasRes.data) ? llamadasRes.data : llamadasRes.data?.data || []; const diagnosticoApis = { leads: leadsRes.diagnostico, historiales: historialesRes.diagnostico, llamadas: llamadasRes.diagnostico, }; // 4. Upsert real: onConflict contra la columna única de cada tabla, en lotes, // revisando el error de cada operación (nada se guarda en silencio). const resultados = await Promise.all([ dataLeads.length > 0 ? upsertEnLotes(supabase, "leads", dataLeads, "codigo_prospecto") : { tabla: "leads", insertados: 0, sinClave: 0, errores: [] }, dataHistoriales.length > 0 ? upsertEnLotes(supabase, "historiales", dataHistoriales, "id") : { tabla: "historiales", insertados: 0, sinClave: 0, errores: [] }, dataLlamadas.length > 0 ? upsertEnLotes(supabase, "llamadas", dataLlamadas, "uniqueid") : { tabla: "llamadas", insertados: 0, sinClave: 0, errores: [] }, ]); const huboFalloDeApi = Object.values(diagnosticoApis).some((d) => !d.ok); const huboErrores = resultados.some((r) => r.errores.length > 0) || huboFalloDeApi; // 5. Respuesta: nunca decimos "success" si algo falló. return new Response( JSON.stringify({ status: huboErrores ? "partial_error" : "success", advertencia_credenciales: usandoAnonFallback ? "SUPABASE_SERVICE_ROLE_KEY no está configurada; se usó SUPABASE_ANON_KEY. Si RLS bloquea escritura para 'anon', los upserts fallarán." : null, registros_recibidos: { leads: dataLeads.length, historiales: dataHistoriales.length, llamadas: dataLlamadas.length, }, diagnostico_apis: diagnosticoApis, resultado_por_tabla: resultados, }), { status: huboErrores ? 207 : 200, // 207 Multi-Status: parte ok, parte falló headers: { ...corsHeaders, "Content-Type": "application/json" }, } ); } catch (err: any) { return new Response( JSON.stringify({ status: "error", message: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" }, } ); } });
